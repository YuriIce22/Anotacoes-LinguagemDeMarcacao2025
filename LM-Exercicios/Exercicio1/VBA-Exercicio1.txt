Sub cartas_personalizadas()

    'Declarando as variaveis'
    Dim objeto_cartas As Object
    Dim arquivo_cartas As Object
    Dim selecao As Object
    Dim ult_linha As Long
    Dim ult_coluna As Long
    Dim linha As Long
    Dim coluna As Long
   
    'Criar OBJETO cartas'
    Set objeto_cartas = CreateObject("Word.Application")
    objeto_cartas.Visible = True
       
    'Descobrir ultimas linhas e colunas'
    ult_linha = Cells(Rows.Count, 1).End(xlUp).Row
    ult_coluna = Cells(1, Columns.Count).End(xlToLeft).Column
   
    'Loop para cada linha dos dados'
    For linha = 2 To ult_linha
        'Abrir modelo'
        Set arquivo_cartas = objeto_cartas.Documents.Open("C:\Users\Aluno\Desktop\Exercicio\Exercicio1\Exercicio1.docx")
        Set selecao = arquivo_cartas.Application.Selection
       
        For coluna = 1 To ult_coluna
            With selecao.Find
                .Text = Cells(1, coluna).Value
                .Replacement.Text = Cells(linha, coluna).Value
                .Execute Replace:=2 'wdReplaceAll = 2'
            End With
        Next coluna
       
       
            'Salvar DOC'
            arquivo_cartas.SaveAs2 "C:\Users\Aluno\Desktop\Exercicio\Exercicio1\Preenchido\CARTAS - " & Cells(linha, 1).Value & " .docx"
            arquivo_cartas.Close False  'Fechar DOC'
            Cells(ult_linha + 1, 1).Value = "Progresso: " & (linha - 1) & "/" & (ult_linha - 1) 'Progresso'
        Next linha
       
            'Apagar progresso'
            Cells(ult_linha + 1, 1).ClearContents
           
            'Fechar Word'
            objeto_cartas.Quit
           
            'Mensagem positiva
            MsgBox "Cartas geradas!"

End Sub
